\begin{example}
Let $\P_1 = \{\{1, 2\}, \{3\}, \{4\}\}$, $\P_2 = \{\{1,3\},\{2\},\{4\}\}$, $\P_1\vee \P_2 = \{\{1,2,3\},\{4\}\}$, $\P_1 \wedge \P_2 = \{\{1\},\{2\},\{3\}, \{4\}\}$
\end{example}
\begin{example}
Let $\P_1 = \{\{1\}, \{2\}, \{3\}\}$, $\P_2 = \{\{1,2\},\{3\}\}$, $\P_1 / \P_2 = \{\{\{1\},\{2\}\},\{\{3\}\}\}$.
\end{example}
